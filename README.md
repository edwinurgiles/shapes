# shapes
import turtle

s = turtle.Turtle()
s.penup()
s.setposition(250,430)
s.begin_fill()
s.forward(75)
s.right(90)
s.forward(75)
s.right(90)
s.forward(75)
s.right(90)
s.forward(75)
s.right(90)
s.color("red")
s.end_fill()
turtle.done()

c = turtle.Turtle()

c.color("purple")
c.end_fill()
c.penup()
turtle.done()
